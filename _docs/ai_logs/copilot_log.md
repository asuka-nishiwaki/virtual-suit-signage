import React, { useEffect, useRef, useState } from 'react';
import * as tf from '@tensorflow/tfjs';
import * as posenet from '@tensorflow-models/posenet';
import * as posedetection from '@tensorflow-models/pose-detection';

interface KeyPoint {
  name: string;
  position: { x: number; y: number };
  score: number;
}

interface BodyMetrics {
  shoulderWidth: number;
  armLength: number;
  neckLength: number;
  legLength: number;
  posture: string;
  bodyType: string;
}

export const PoseEstimation: React.FC<{
  onDiagnosisComplete: (metrics: BodyMetrics) => void;
}> = ({ onDiagnosisComplete }) => {
  const videoRef = useRef<HTMLVideoElement>(null);
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const [direction, setDirection] = useState<0 | 90 | 180 | 270>(0);
  const [capturedPoses, setCapturedPoses] = useState<any[]>([]);
  const detectorRef = useRef<any>(null);

  // 初期化: MediaPipe Pose Detectorを読み込む
  useEffect(() => {
    const initDetector = async () => {
      await tf.ready();
      const detector = await posedetection.createDetector(
        posedetection.SupportedModels.MoveNet,
        {
          modelType: posedetection.movenet.modelType.MULTIPOSE_LIGHTNING,
        }
      );
      detectorRef.current = detector;
    };

    initDetector();
  }, []);

  // カメラ初期化
  useEffect(() => {
    const initCamera = async () => {
      try {
        const stream = await navigator.mediaDevices.getUserMedia({
          video: {
            width: { ideal: 1280 },
            height: { ideal: 720 },
            facingMode: 'user',
          },
        });
        if (videoRef.current) {
          videoRef.current.srcObject = stream;
        }
      } catch (error) {
        console.error('カメラアクセスエラー:', error);
      }
    };

    initCamera();
  }, []);

  // リアルタイムポーズ推定
  useEffect(() => {
    let animationId: number;

    const detectPose = async () => {
      if (videoRef.current && detectorRef.current && canvasRef.current) {
        const poses = await detectorRef.current.estimatePoses(videoRef.current);
        drawPoseOnCanvas(poses[0]);
        animationId = requestAnimationFrame(detectPose);
      }
    };

    detectPose();

    return () => cancelAnimationFrame(animationId);
  }, []);

  // キャンバスにポーズを描画
  const drawPoseOnCanvas = (pose: any) => {
    const canvas = canvasRef.current;
    if (!canvas || !pose) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // キーポイント描画
    pose.keypoints.forEach((point: any) => {
      if (point.score > 0.5) {
        ctx.fillStyle = '#00ff00';
        ctx.beginPath();
        ctx.arc(point.x, point.y, 5, 0, 2 * Math.PI);
        ctx.fill();
      }
    });

    // 骨を描画
    drawSkeleton(pose.keypoints, ctx);
  };

  // スケルトン描画（骨の接続）
  const drawSkeleton = (keypoints: any[], ctx: CanvasRenderingContext2D) => {
    const pairs = [
      [5, 6], // 肩
      [5, 7], // 左肘
      [7, 9], // 左手首
      [6, 8], // 右肘
      [8, 10], // 右手首
      [5, 11], // 左腰
      [6, 12], // 右腰
      [11, 13], // 左膝
      [13, 15], // 左足首
      [12, 14], // 右膝
      [14, 16], // 右足首
    ];

    ctx.strokeStyle = '#00ff00';
    ctx.lineWidth = 2;

    pairs.forEach(([from, to]) => {
      const fromPoint = keypoints[from];
      const toPoint = keypoints[to];

      if (fromPoint.score > 0.5 && toPoint.score > 0.5) {
        ctx.beginPath();
        ctx.moveTo(fromPoint.x, fromPoint.y);
        ctx.lineTo(toPoint.x, toPoint.y);
        ctx.stroke();
      }
    });
  };

  // 4方向撮影ボタン処理
  const captureDirection = async () => {
    if (videoRef.current && detectorRef.current) {
      const poses = await detectorRef.current.estimatePoses(videoRef.current);
      setCapturedPoses([...capturedPoses, { direction, pose: poses[0] }]);

      // 次の方向へ
      const directions: (0 | 90 | 180 | 270)[] = [0, 90, 180, 270];
      const nextIdx = directions.indexOf(direction) + 1;

      if (nextIdx < 4) {
        setDirection(directions[nextIdx]);
      } else {
        // 診断実行
        const metrics = calculateBodyMetrics(capturedPoses);
        onDiagnosisComplete(metrics);
      }
    }
  };

  // 体型メトリクス計算
  const calculateBodyMetrics = (poses: any[]): BodyMetrics => {
    const frontPose = poses[0]?.pose;
    if (!frontPose) return {} as BodyMetrics;

    const kp = frontPose.keypoints;

    // 肩幅 = 左肩(5)と右肩(6)の距離
    const shoulderWidth = Math.hypot(
      kp[5].x - kp[6].x,
      kp[5].y - kp[6].y
    );

    // 腕の長さ = 肩(5)と手首(9)の距離
    const armLength = Math.hypot(
      kp[5].x - kp[9].x,
      kp[5].y - kp[9].y
    );

    // 首の長さ = 肩(5)と鼻(0)の距離
    const neckLength = Math.hypot(
      kp[5].x - kp[0].x,
      kp[5].y - kp[0].y
    );

    // 脚の長さ = 腰(11)と足首(15)の距離
    const legLength = Math.hypot(
      kp[11].x - kp[15].x,
      kp[11].y - kp[15].y
    );

    // 姿勢判定（肩の傾き）
    const shoulderSlope = Math.abs(kp[5].y - kp[6].y);
    const posture = shoulderSlope < 20 ? 'なで肩' : '肩幅広';

    // 体型分類（腕と脚の比率）
    const ratio = armLength / legLength;
    const bodyType = ratio > 0.5 ? '標準型' : '長脚型';

    return {
      shoulderWidth,
      armLength,
      neckLength,
      legLength,
      posture,
      bodyType,
    };
  };

  return (
    <div className="pose-estimation-container">
      <div className="instruction">
        <h2>方向を選択してください: {direction}°</h2>
        <p className="direction-text">
          {direction === 0 && '正面を向いてください'}
          {direction === 90 && '左を向いてください'}
          {direction === 180 && '後ろを向いてください'}
          {direction === 270 && '右を向いてください'}
        </p>
      </div>

      <div className="video-container">
        <video
          ref={videoRef}
          autoPlay
          playsInline
          style={{ display: 'none' }}
        />
        <canvas
          ref={canvasRef}
          width={1280}
          height={720}
          className="pose-canvas"
        />
        <div className="frame-guide">
          {/* ガイドフレーム: ユーザーが体をこの枠に合わせる */}
        </div>
      </div>

      <button onClick={captureDirection} className="capture-btn">
        {capturedPoses.length === 3 ? '診断実行' : '次へ'}
      </button>
    </div>
  );
};

```

import numpy as np
from PIL import Image
import tensorflow as tf
from tensorflow import keras
from typing import Dict, List

class BodyDiagnosticAI:
    def __init__(self):
        # 骨格分類モデル (TensorFlow)
        self.skeleton_model = keras.models.load_model('models/skeleton_classifier.h5')
        # 体型分類モデル
        self.body_type_model = keras.models.load_model('models/body_type_classifier.h5')
        # パーソナルカラー分析モデル
        self.color_model = keras.models.load_model('models/personal_color_classifier.h5')

    def analyze_skeleton(self, keypoints: np.ndarray) -> str:
        """
        骨格タイプ推定: ナチュラル / ストレート / ウェーブ
        """
        # キーポイントを正規化
        normalized = self._normalize_keypoints(keypoints)
        
        # 骨格特徴を抽出
        features = self._extract_skeleton_features(normalized)
        
        # 分類
        prediction = self.skeleton_model.predict(features)[0]
        skeleton_types = ['ナチュラル', 'ストレート', 'ウェーブ']
        
        return skeleton_types[np.argmax(prediction)]

    def analyze_body_type(self, metrics: Dict) -> str:
        """
        体型分類: 標準型 / 細身型 / グラマー型
        """
        bmi = metrics['weight'] / (metrics['height'] ** 2)
        aspect_ratio = metrics['shoulderWidth'] / metrics['legLength']
        
        if bmi < 20:
            return '細身型'
        elif bmi < 25:
            return '標準型'
        else:
            return 'グラマー型'

    def analyze_personal_color(self, face_image: Image) -> Dict:
        """
        パーソナルカラー分析: 顔色から色温度を判定
        ブルベ冬 / ブルベ秋 / イエベ春 / イエベ秋
        """
        # 顔画像を処理
        face_array = np.array(face_image.resize((224, 224))) / 255.0
        
        # 色温度分析
        color_prediction = self.color_model.predict(np.expand_dims(face_array, 0))[0]
        color_types = ['ブルベ冬', 'ブルベ秋', 'イエベ春', 'イエベ秋']
        
        color_type = color_types[np.argmax(color_prediction)]
        confidence = float(np.max(color_prediction))
        
        # 推奨色
        color_recommendations = {
            'ブルベ冬': ['ネイビー', 'ブラック', 'ホワイト', 'シルバー'],
            'ブルベ秋': ['ネイビー', 'グレー', 'ブラウン', 'ゴールド'],
            'イエベ春': ['ベージュ', 'キャメル', 'ライトグレー', 'ゴールド'],
            'イエベ秋': ['オリーブ', 'テラコッタ', 'キャメル', 'ゴールド'],
        }
        
        return {
            'colorType': color_type,

import * as THREE from 'three';

export class ClothSimulationService {
  private particles: Particle[] = [];
  private constraints: Constraint[] = [];
  private gravity = new THREE.Vector3(0, -0.98, 0);
  private damping = 0.99;
  private constraintIterations = 3;

  class Particle {
    position: THREE.Vector3;
    previousPosition: THREE.Vector3;
    pinned: boolean = false;
    mass: number = 1;

    constructor(x: number, y: number, z: number) {
      this.position = new THREE.Vector3(x, y, z);
      this.previousPosition = this.position.clone();
    }

    applyForce(force: THREE.Vector3) {
      force.divideScalar(this.mass);
      this.position.add(force);
    }

    update() {
      const vel = this.position.clone().sub(this.previousPosition);
      vel.multiplyScalar(this.damping);
      this.previousPosition.copy(this.position);
      this.position.add(vel);
      this.position.add(this.gravity);
    }
  }

  class Constraint {
    p1: Particle;
    p2: Particle;
    restDistance: number;

    constructor(p1: Particle, p2: Particle) {
      this.p1 = p1;
      this.p2 = p2;
      this.restDistance = p1.position.distanceTo(p2.position);
    }

    solve() {
      const delta = this.p2.position.clone().sub(this.p1.position);
      const currentDistance = delta.length();
      const difference = (this.restDistance - currentDistance) / currentDistance;

      const offset = delta.multiplyScalar(difference * 0.5);

      this.p1.position.sub(offset);
      this.p2.position.add(offset);
    }
  }

  createCloth(width: number, height: number, clothGeometry: THREE.Geometry) {
    // 布をパーティクルと制約で構成
    const clothSegments = 12;
    let index = 0;

    for (let v = 0; v <= height; v++) {
      for (let u = 0; u <= width; u++) {
        const particle = new Particle(
          u / width,
          1 - v / height,
          0
        );

        if (v === 0 && (u === 0 || u === width)) {
          particle.pinned = true; // 肩で固定
        }

        this.particles.push(particle);
        index++;
      }
    }

    // 制約追加 (布の構造)
    for (let v = 0; v < height; v++) {
      for (let u = 0; u < width; u++) {
        const idx1 = v * (width + 1) + u;
        const idx2 = idx1 + 1;
        const idx3 = idx1 + width + 1;

        this.constraints.push(new Constraint(this.particles[idx1], this.particles[idx2]));
        this.constraints.push(new Constraint(this.particles[idx1], this.particles[idx3]));
      }
    }
  }

  simulate() {
    // パーティクル更新
    this.particles.forEach((p) => {
      if (!p.pinned) {
        p.update();
      }
    });

    // 制約解決
    for (let iter = 0; iter < this.constraintIterations; iter++) {
      this.constraints.forEach((c) => c.solve());
    }
  }

  // 腕の動きに布を追従させる
  followArmMotion(shoulderPos: THREE.Vector3, elbowPos: THREE.Vector3) {
    // 肩と肘の位置に基づいて布を変形
    this.particles.forEach((p) => {
      if (p.pinned) {
        p.position.copy(shoulderPos);
      }
    });
  }

  getGeometry(): THREE.BufferGeometry {
    const positions = [];
    this.particles.forEach((p) => {
      positions.push(p.position.x, p.position.y, p.position.z);
    });

    const geometry = new THREE.BufferGeometry();
    geometry.setAttribute(
      'position',
      new THREE.BufferAttribute(new Float32Array(positions), 3)
    );

    return geometry;
  }
}

  const [completedSteps, setCompletedSteps] = useState<Set<string>>(new Set());

  const handleDiagnosisComplete = async (metrics: BodyMetrics) => {

        {currentScreen === 'diagnosis' && (
          <div className="screen-wrapper">
            <PoseEstimation
              onDiagnosisComplete={handleDiagnosisComplete}
              onSkipDiagnosis={handleSkipDiagnosis}
            />
          </div>
        )}

        {currentScreen === 'result' && diagnosisData && (
          <div className="screen-wrapper">
            <DiagnosisResult
              data={diagnosisData}
              onStartVirtualTryOn={handleStartVirtualTryOn}
              onSkipDiagnosis={handleSkipDiagnosis}
            />
          </div>
        )}

        {currentScreen === 'virtual-tryon' && (
          <div className="screen-wrapper">
            <div className="placeholder-screen">
              <h2>🎨 バーチャル試着画面</h2>
              <p>（VirtualTryOn コンポーネントをここに配置）</p>
              <div className="button-group">
                <button onClick={handleShowImpressionAnalysis} className="btn btn-secondary">
                  印象診断を見る
                </button>
                <button onClick={handleShowSceneAppropriations} className="btn btn-secondary">
                  シーン適正を見る
                </button>
                <button onClick={handleReset} className="btn btn-primary">
                  最初からやり直す
                </button>
              </div>
            </div>
          </div>
        )}

        {currentScreen === 'impression' && (
          <div className="screen-wrapper">
            <div className="placeholder-screen">
              <h2>✨ 印象診断</h2>
              <p>（ImpressionAnalysis コンポーネントをここに配置）</p>
              <button onClick={handleTryAnotherSuit} className="btn btn-secondary">
                戻る
              </button>
            </div>
          </div>
        )}

        {currentScreen === 'scene' && (
          <div className="screen-wrapper">
            <div className="placeholder-screen">
              <h2>🎭 シーン適正</h2>
              <p>（SceneAppropriations コンポーネントをここに配置）</p>
              <button onClick={handleTryAnotherSuit} className="btn btn-secondary">
                戻る
              </button>
            </div>
          </div>
        )}
      </main>

      {/* フッター - 進捗インジケーター */}
      <footer className="app-footer">
        <div className="progress-indicator">
          <div className={`step ${completedSteps.has('pose-estimation') ? 'completed' : ''}`}>
            <span className="step-number">①</span>
            <span className="step-label">撮影</span>
          </div>
          <div className={`step ${completedSteps.has('diagnosis-result') ? 'completed' : ''}`}>
            <span className="step-number">②</span>
            <span className="step-label">診断</span>
          </div>
          <div className={`step ${currentScreen === 'virtual-tryon' ? 'active' : ''}`}>
            <span className="step-number">③</span>
            <span className="step-label">試着</span>
          </div>
        </div>
      </footer>
    </div>
  );
}

export default App;
